import matplotlib.pyplot as plt

fig, ax = plt.subplots()

ax.plot([1, 2, 3], label='One breath')
ax.plot([3, 2, 1], label='The other')

legend = ax.legend(loc='best')

# Set vertical alignment and rotation
for text in legend.get_texts():
    text.set_rotation(90)
    text.set_ha('left')   # or 'right' or 'center'
    text.set_va('center') # keeps text centered vertically

plt.tight_layout()
plt.show()
