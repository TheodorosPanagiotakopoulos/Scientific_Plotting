import os
import re

def get_sorted_group_files(directory):
    pattern = re.compile(r"default_group_(\d+)\.txt")
    matched_files = []

    for filename in os.listdir(directory):
        match = pattern.match(filename)
        if match:
            number = int(match.group(1))
            matched_files.append((number, filename))

    # Sort by the extracted number
    matched_files.sort()
    # Return only the filenames in sorted order
    return [filename for _, filename in matched_files]
