import re

# List to store the extracted numbers
numbers = []

# Open the file for reading
with open('your_file.txt', 'r') as file:
    for line in file:
        # Check if "Elapsed" is in the line and contains the pattern "done(number out of"
        if "Elapsed" in line and "done(" in line and " out of" in line:
            # Use regular expression to find the number after done(
            match = re.search(r'done\((\d+) out of', line)
            if match:
                numbers.append(match.group(1))

# Print the list of numbers
print(numbers)
