import os
import shutil

def reset_folder(path, folder_name):
    folder_path = os.path.join(path, folder_name)
    
    # If folder exists, delete it
    if os.path.exists(folder_path):
        shutil.rmtree(folder_path)
    
    # Recreate folder
    os.makedirs(folder_path)

    return folder_path
