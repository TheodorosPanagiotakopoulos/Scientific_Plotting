import matplotlib.pyplot as plt

def plot_dataframe(df):
    # Check if the DataFrame is empty
    if df.empty:
        print("The DataFrame is empty. Nothing to plot.")
        return

    # Ensure the index is numeric for division
    try:
        x_values = df.index.astype(float) / 1000
    except ValueError:
        print("Index must be numeric to divide by 1000.")
        return

    # Plot all columns
    plt.figure(figsize=(10, 6))
    for column in df.columns:
        plt.plot(x_values, df[column], label=column)

    # Labels and title
    plt.xlabel("Steps × 10⁻³")
    plt.ylabel("Values")
    plt.title("DataFrame Column Plot")
    plt.legend()
    plt.grid(True)
    plt.tight_layout()
    plt.show()
