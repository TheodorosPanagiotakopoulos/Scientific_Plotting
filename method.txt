import pandas as pd

# List to store DataFrames
dfs = []

for i in range(5):
    col1 = 'group_' + str(i)
    col2 = 'Rmse_' + str(i)
    
    df = pd.DataFrame({
        col1: ['A'],
        col2: [0.1 * i]
    })
    dfs.append(df)

# Concatenate all DataFrames horizontally
final_df = pd.concat(dfs, axis=1)
