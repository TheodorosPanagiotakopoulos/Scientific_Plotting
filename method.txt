import pandas as pd

# Assume big_dict is your nested dictionary
# Format: { 'key_xxxx': { 'df_name': dataframe, ... }, ... }

result_df = pd.DataFrame()

for outer_key, inner_dict in big_dict.items():
    suffix = outer_key.split("_")[-1]  # Get the 'xxxx' from 'key_xxxx'
    
    for df_name, df in inner_dict.items():
        if "_0.00" in df_name:
            if "max_aitable" in df.columns:
                # Create the new column name
                new_col_name = f"max_aitable_{suffix}"
                
                # Add to the result dataframe
                result_df[new_col_name] = df["max_aitable"].reset_index(drop=True)

# Now result_df contains the desired merged columns
