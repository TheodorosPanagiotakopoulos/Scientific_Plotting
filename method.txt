import pandas as pd

def add_model_difference_column(df: pd.DataFrame) -> pd.DataFrame:
	"""
	Adds a new column 'Model_CD_minus_Rig_Model_CD' to the DataFrame
	which is the result of Model_CD - Rig_Model_CD.

	Parameters:
		df (pd.DataFrame): The input DataFrame with columns 'Model_CD' and 'Rig_Model_CD'

	Returns:
		pd.DataFrame: The updated DataFrame with the new column appended
	"""
	if 'Model_CD' not in df.columns or 'Rig_Model_CD' not in df.columns:
		raise ValueError("Input DataFrame must contain 'Model_CD' and 'Rig_Model_CD' columns")

	df['Model_CD_minus_Rig_Model_CD'] = df['Model_CD'] - df['Rig_Model_CD']
	return df
