import os

def distribute_tmp_files_to_default_groups(default_groups_path, tmp_folder):
	# Get all .txt files in tmp_folder
	tmp_files = sorted([
		os.path.join(tmp_folder, file_name)
		for file_name in os.listdir(tmp_folder)
		if file_name.endswith(".txt") and os.path.isfile(os.path.join(tmp_folder, file_name))
	])

	# Prepare paths for default_group_1.txt to default_group_7.txt
	group_files = []
	for i in range(1, 8):
		group_file = os.path.join(default_groups_path, "default_group_" + str(i) + ".txt")
		group_files.append(group_file)

	group_index = 0
	for tmp_file in tmp_files:
		with open(tmp_file, 'r') as tf:
			content = tf.read()

		with open(group_files[group_index], 'a') as gf:
			gf.write(content)

		print("Appended '" + tmp_file + "' to '" + group_files[group_index] + "'")

		group_index = (group_index + 1) % 7
