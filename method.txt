import pandas as pd
import re

def split_dataframe_by_number(df):
    # Dictionary to hold grouped columns
    grouped_cols = {}

    # Extract number from each column name and group
    for col in df.columns:
        match = re.match(r"slit_([+-]?\d*\.?\d+)_.*", col)
        if match:
            number = match.group(1)
            grouped_cols.setdefault(number, []).append(col)
        else:
            print(f"Skipping column with unexpected name format: {col}")

    # Iterate and print dataframes
    for number in sorted(grouped_cols, key=lambda x: float(x)):
        print()  # Blank line before each group
        print(f"Data for slit number {number}:\n")
        print(df[grouped_cols[number]])
