import math

def one_order_smaller_than_extremes(lst):
    max_val = max(lst)
    min_val = min(lst)

    def one_order_less(n):
        if n == 0:
            return 0
        power = math.floor(math.log10(abs(n))) - 1
        return 10 ** power

    return (one_order_less(max_val), one_order_less(min_val))
