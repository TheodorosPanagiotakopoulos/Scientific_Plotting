import pandas as pd
import numpy as np

def mark_max_on_top_and_print_row(df: pd.DataFrame, column_name: str, new_column_name: str) -> pd.DataFrame:
	"""
	Adds a new column with the max value of a specified column in the first row.
	Prints the row where the max value is found.

	Parameters:
		df (pd.DataFrame): The input DataFrame.
		column_name (str): Column to compute the max from.
		new_column_name (str): Name for the new column.

	Returns:
		pd.DataFrame: Updated DataFrame with the new column.
	"""
	if column_name not in df.columns:
		raise ValueError(f"Column '{column_name}' not found in DataFrame")

	max_val = df[column_name].max()
	max_rows = df[df[column_name] == max_val]

	print("Row(s) where max is found:")
	print(max_rows)

	new_col = [max_val] + [np.nan] * (len(df) - 1)
	df[new_column_name] = new_col
	return df
