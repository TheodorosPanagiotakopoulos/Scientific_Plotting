import matplotlib.pyplot as plt
import numpy as np

def plot_trend(data, degree=1):
    """
    Plot a trend line for the given data.

    Parameters:
    - data (list or array): Numeric values
    - degree (int): Degree of the polynomial trend (default = 1 for linear)
    """
    x = np.arange(len(data))
    z = np.polyfit(x, data, degree)
    p = np.poly1d(z)
    
    plt.plot(x, p(x), label=f'{degree}-degree Trend', color='red', linewidth=2)
    plt.title(f'{degree}-Degree Trend of Data')
    plt.xlabel('Index')
    plt.ylabel('Value')
    plt.legend()
    plt.grid(True)
    plt.show()
