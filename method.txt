import pandas as pd
import numpy as np

# Example DataFrame
df = pd.DataFrame({
    'column': [np.nan, np.nan, 42, np.nan, np.nan]
})

# Get the non-NaN value
non_nan_value = df['column'].dropna().iloc[0]  # assumes only one non-NaN value

# Create a new column with that value at the top and the rest as NaN
df['column'] = [non_nan_value] + [np.nan] * (len(df) - 1)

print(df)
