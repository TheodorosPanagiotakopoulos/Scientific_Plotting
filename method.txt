def replace_cal_ver(file_path, choice):
	if choice not in ("CAL_TO_VER", "VER_TO_CAL"):
		print("Invalid choice. Use 'CAL_TO_VER' or 'VER_TO_CAL'.")
		return

	with open(file_path, 'r') as f:
		lines = f.readlines()

	updated_lines = []
	for line in lines:
		if choice == "CAL_TO_VER":
			updated_lines.append(line.replace("CAL", "VER"))
		elif choice == "VER_TO_CAL":
			updated_lines.append(line.replace("VER", "CAL"))

	with open(file_path, 'w') as f:
		f.writelines(updated_lines)

	print(f"File '{file_path}' updated with choice: {choice}")
