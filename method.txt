def sort_slit_columns(df, prefix="slit_"):
    """
    Reorders DataFrame columns by the numeric part after a given prefix.

    Assumes all column names starting with the prefix are in the format: prefix<number>,
    e.g., 'slit_-2.5', 'slit_0', 'slit_1'.

    Args:
        df (pd.DataFrame): Input DataFrame.
        prefix (str): Prefix to identify target columns.

    Returns:
        pd.DataFrame: DataFrame with slit columns ordered numerically, others after.
    """
    # Split columns into matching and non-matching
    slit_cols = [col for col in df.columns if col.startswith(prefix)]
    other_cols = [col for col in df.columns if not col.startswith(prefix)]

    # Sort matching columns by the numeric part
    slit_cols_sorted = sorted(slit_cols, key=lambda col: float(col[len(prefix):]))

    # Combine sorted + other columns
    final_cols = slit_cols_sorted + other_cols
    return df[final_cols]
