#!/bin/bash

# Usage: ./update_theta.sh input_file new_number

if [ $# -ne 2 ]; then
  echo "Usage: $0 input_file new_number"
  exit 1
fi

input_file="$1"
new_number="$2"

# Extract and print original numbers
echo "Original <theta> values:"
grep -oP '(?<=<theta>)-?[0-9]+(\.[0-9]+)?(?=</theta>)' "$input_file"

# Replace with the new number
sed -i -E "s|<theta>-?[0-9]+(\.[0-9]+)?</theta>|<theta>${new_number}</theta>|g" "$input_file"

echo "All <theta> values updated to: $new_number"
