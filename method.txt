def find_closest(val, candidates):
    return min(candidates, key=lambda x: abs(x - val))

def match_elements(small_list, large_list):
    result = []
    for item in small_list:
        if item in large_list:
            result.append(item)
        else:
            closest = find_closest(item, large_list)
            result.append(closest)
    return result
