#!/bin/bash

# Hardcoded file path
file="path/to/your/file.txt"

# Hardcoded new number
new_number="-2.71"  # <- Change this to whatever number you want

# Print current <theta> values
echo "Original <theta> values in $file:"
grep -oP '(?<=<theta>)-?[0-9]+(\.[0-9]+)?(?=</theta>)' "$file"

# Replace all <theta> values with the new number
sed -i -E "s|<theta>-?[0-9]+(\.[0-9]+)?</theta>|<theta>${new_number}</theta>|g" "$file"

echo "All <theta> values updated to: $new_number"
