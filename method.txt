import re

def get_loss_numbers(filename):
    loss_numbers = []
    
    with open(filename, 'r') as file:
        for line in file:
            if "Elapsed" in line and "out of" in line and "loss0=" in line:
                match = re.search(r'loss0=([\d.]+)', line)
                if match:
                    loss_numbers.append(float(match.group(1)))
    
    if not loss_numbers:
        print("Error: No 'loss0=' values found in lines with 'Elapsed' and 'out of'.")
    else:
        print("Extracted loss values:", loss_numbers)

# Execute
if __name__ == "__main__":
    get_loss_numbers("status.log")
