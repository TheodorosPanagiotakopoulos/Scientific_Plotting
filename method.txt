# main.py

from concurrent.futures import ProcessPoolExecutor
from tasks import func1, func2, func3

if __name__ == "__main__":
    with ProcessPoolExecutor() as executor:
        futures = [
            executor.submit(func1, 4),                 # func1(4)
            executor.submit(func2, 10, 20),            # func2(10, 20)
            executor.submit(func3, "hello world")      # func3("hello world")
        ]
        results = [f.result() for f in futures]
        print("Results:", results)
