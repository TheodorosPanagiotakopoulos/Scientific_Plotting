def update_ver_to_cal(filepath):
    with open(filepath, 'r') as file:
        lines = file.readlines()

    updated_lines = []
    for line in lines:
        if 'anchor' in line.lower() and 'VER' in line:
            updated_lines.append(line.replace('VER', 'CAL'))
        else:
            updated_lines.append(line)

    with open(filepath, 'w') as file:
        file.writelines(updated_lines)
