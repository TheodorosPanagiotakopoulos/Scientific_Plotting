import pandas as pd

def sort_dataframe_by_index_number(df: pd.DataFrame) -> pd.DataFrame:
	"""
	Sort DataFrame rows by the numeric part of the index formatted as 'modelchk_<number>_NC'.
	"""
	# Extract numbers as integers
	numbers = df.index.to_series().str.extract(r'modelchk_(\d+)_NC')[0].astype(int)

	# Combine with original DataFrame and sort by number
	df_sorted = df.copy()
	df_sorted['__num__'] = numbers
	df_sorted = df_sorted.sort_values('__num__')
	df_sorted = df_sorted.drop(columns='__num__')

	return df_sorted
