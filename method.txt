def add_max_column(df, col):
    df = df.copy()
    max_value = df[col].max()
    df['max'] = np.nan
    max_index = df[col].idxmax()
    df.at[max_index, 'max'] = max_value
    
    # Move max row to the top (position 0), then reset index to start from 1
    df = pd.concat([df.loc[[max_index]], df.drop(index=max_index)])
    df.index = range(1, len(df) + 1)  # Reindex to start at 1
    return df
