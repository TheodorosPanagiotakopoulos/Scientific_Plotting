import pandas as pd

def sort_dataframe_by_index_number(df: pd.DataFrame) -> pd.DataFrame:
	"""
	Sorts a DataFrame based on the numerical value in the index formatted as 'modelchk_number_NC'.

	Parameters:
		df (pd.DataFrame): The input DataFrame with index of the form 'modelchk_number_NC'.

	Returns:
		pd.DataFrame: A DataFrame sorted by the extracted numerical value.
	"""
	# Extract the number from the index
	df['_sort_key'] = df.index.str.extract(r'modelchk_(\d+)_NC')[0].astype(int)
	
	# Sort by the extracted number
	df_sorted = df.sort_values('_sort_key').drop(columns=['_sort_key'])

	return df_sorted
