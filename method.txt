import os

def check_or_scp_copy(src_folder_path):
    folder_name = os.path.basename(os.path.normpath(src_folder_path))
    dest_folder_path = os.path.join(os.getcwd(), folder_name)

    if os.path.exists(dest_folder_path):
        print("The folder '" + folder_name + "' already exists in the current directory.")
    else:
        scp_command = "scp -rp " + src_folder_path + " " + os.getcwd()
        result = os.system(scp_command)
        if result == 0:
            print("Folder '" + folder_name + "' copied successfully to current directory using SCP.")
        else:
            print("Failed to copy folder '" + folder_name + "' using SCP.")
